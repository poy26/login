Git clone https://github.com/poy26/login.git

cd login

nano login.sh ( ubah username )

nano password.sh ( ubah password )

mv login.sh $HOME

mv password.sh $HOME

nano ../usr/etc/bash.bashrc

apabila memakai zshrc

nano ./zshrc

ketik kolom paling bawah login.sh dan password.sh
